15 openfsl::fat32::FormatOptions
16  openfsl::fat32::getDefaultFormatOptions(const lba48_t fsSize) {
17  FormatOptions ret;
18  ret.volumeLabel = "NO NAME";
19 
20  auto tpoint = std::chrono::system_clock::now();
21  time_t ttenth = std::chrono::duration_cast<std::chrono::seconds>
22  (tpoint.time_since_epoch()).count() / 10;
23  ttenth %= 100;
24  time_t t = std::chrono::system_clock::to_time_t(tpoint);
25  std::tm currentTime = fsl_localtime(t);
26 
27  uint16_t tmp1 = (((uint8_t)currentTime.tm_mon + 1) << 8) +
28  (uint8_t)currentTime.tm_mday;
29  uint16_t tmp2 = ((uint8_t)currentTime.tm_sec << 8) + (uint8_t)ttenth;
30  ret.volumeSerial = (tmp1 + tmp2) << 16;
31 
32  tmp1 = ((uint8_t)currentTime.tm_hour << 8) + (uint8_t)currentTime.tm_min;
33  tmp2 = (uint16_t)currentTime.tm_year + 1900;
34  ret.volumeSerial += tmp1 + tmp2;
35 
36  std::cout << std::hex << ret.volumeSerial << std::dec << std::endl;
37 
38  ret.biosDriveNum = 0x80;
39 
40  if (fsSize < 532480)
41  ret.sectorsPerCluster = 1;
42  else if (fsSize < 16777216)
43  ret.sectorsPerCluster = 8;
44  else if (fsSize < 33554432)
45  ret.sectorsPerCluster = 16;
46  else if (fsSize < 67108864)
47  ret.sectorsPerCluster = 32;
48  else
49  ret.sectorsPerCluster = 64;
50 
51  ret.backupBSLocation = 6;
52  ret.fsinfoLocation = 1;
53  ret.reservedSectorCount = 32;
54 
55  ret.fatAreaCount = 2;
56  ret.fatMediaType = 0xF8;
57  ret.rootDirectoryCluster = 2;
58 
59  return ret;
60 }

62 error_t openfsl::fat32::FAT32::format(const lba48_t offset,
63  const lba48_t size,
64  const FormatOptions formatOptions,
65  const bool fastFormat) {
66  BPBSector bpb;
67  FSINFOSector fsinfo;
68 
69  memset(&bpb, 0, sizeof(BPBSector));
70  memset(&fsinfo, 0, sizeof(FSINFOSector));
71 
72  // Set signatures
73  bpb.bpbVBRSignature = 0xAA55;
74  bpb.ebpbExtendedBootSign = 0x29;
75 
76  fsinfo.fsinfoSignature1 = 0x41615252;
77  fsinfo.fsinfoSignature2 = 0x61417272;
78  fsinfo.fsinfoSignature3 = 0xAA55;
79 
80  // Copy VBR boot codes
81  bpb.bpbJumpCode[0] = 0xEB;
82  bpb.bpbJumpCode[1] = 0x58;
83  bpb.bpbJumpCode[2] = 0x90;
84 
85  memcpy(bpb.bpbBootCode, _blob_vbrboot_bin_data, _blob_vbrboot_bin_len);
86 
87  // Copy volume OEM ID, Label, Volume ID, FS Type
88  if (formatOptions.volumeLabel.length() > 11) {
89  return OPENFSL_ERROR_TOO_LONG_VOLUME_LABEL;
90  }
91 
92  memcpy(bpb.bpbOEMName, "OPENFSL ", 8);
93 
94  memset(bpb.ebpbVolumeLabel, ' ', 11);
95  memcpy(bpb.ebpbVolumeLabel,
96  formatOptions.volumeLabel.c_str(),
97  formatOptions.volumeLabel.length());
98 
99  bpb.ebpbVolumeSerial = (uint32_t)formatOptions.volumeSerial;
100 
101  memcpy(bpb.ebpbFSType, "FAT32 ", 8);
102 
103  // Disk informations
104  if (formatOptions.biosDriveNum < 80) {
105  return OPENFSL_ERROR_FLOPPY_NOT_SUPPORTED;
106  }
107 
108  bpb.bpbMedia = 0xF8;
109  bpb.ebpbPhysicalDriveNum = formatOptions.biosDriveNum;
110  bpb.bpbBytesPerSector =
111  static_cast<uint16_t>(iod->getDiskParameter().bytesPerSector);
112  bpb.bpbSectorsPerTrack =
113  static_cast<uint16_t>(iod->getDiskParameter().sectorPerTrack);
114  bpb.bpbDiskHeads =
115  static_cast<uint16_t>(iod->getDiskParameter().headPerCylinder);
116 
117  // Cluster, sector informations
118  bpb.bpbHiddenSectors = (uint32_t)offset;
119  bpb.bpbTotalSector32 = (uint32_t)size;
120 
121  bpb.bpbSectorPerCluster = formatOptions.sectorsPerCluster;
122  bpb.bpbReservedSectors = (uint16_t)formatOptions.reservedSectorCount;
123 
124  if ((formatOptions.backupBSLocation >= formatOptions.reservedSectorCount) ||
125  (formatOptions.fsinfoLocation >= formatOptions.reservedSectorCount)) {
126  return OPENFSL_ERROR_INVALID_FORMAT_OPTIONS;
127  }
128 
129  bpb.ebpbBSBackupSector = (uint16_t)formatOptions.backupBSLocation;
130  bpb.ebpbFSInfoSector = (uint16_t)formatOptions.fsinfoLocation;
131 
132  // FAT Informations
133  lba48_t tmpval1 = size - formatOptions.reservedSectorCount;
134  lba48_t tmpval2 =
135  128 * formatOptions.sectorsPerCluster + formatOptions.fatAreaCount;
136 
137  bpb.bpbFATCount = formatOptions.fatAreaCount;
138  bpb.ebpbSectorsPerFAT32 = (uint32_t)((tmpval1 + tmpval2 - 1) / tmpval2);
139  bpb.ebpbRootDirectoryCluster = formatOptions.rootDirectoryCluster;
140 
141  cluster_t clusterCount = static_cast<cluster_t>
142  ((size - (formatOptions.reservedSectorCount +
143  bpb.ebpbSectorsPerFAT32 * formatOptions.fatAreaCount))
144  / formatOptions.sectorsPerCluster);
145 
146  if (bpb.ebpbRootDirectoryCluster >= clusterCount)
147  return OPENFSL_ERROR_INVALID_FORMAT_OPTIONS;
148 
149  fsinfo.fsinfoFreeCluster = clusterCount - 1;
150  if (bpb.ebpbRootDirectoryCluster == 1)
151  fsinfo.fsinfoNextFree = 2;
152  else
153  fsinfo.fsinfoNextFree = 0;
154 
155  error_t result;
156 
157  // Zerofill disk
158  if (!fastFormat) {
159  char* emptySector = new char[iod->getDiskParameter().bytesPerSector];
160  memset(emptySector, 0, iod->getDiskParameter().bytesPerSector);
161 
162  for (lba48_t i = offset + formatOptions.reservedSectorCount +
163  bpb.ebpbSectorsPerFAT32 * formatOptions.fatAreaCount;
164  i < offset + size;
165  i++) {
166  result = iod->writeSector(emptySector, i, 1);
167  if (result)
168  return result;
169  }
170 
171  delete[] emptySector;
172  }
173 
174  // Disk operations
175  result = iod->writeSector(&bpb, offset, 1);
176  if (result)
177  return result;
178  result = iod->writeSector(&fsinfo,
179  offset + formatOptions.fsinfoLocation, 1);
180  if (result)
181  return result;
182 
183  // FAT Area initialization
184  Sector fatArea(bpb.ebpbSectorsPerFAT32,
185  iod->getDiskParameter().bytesPerSector);
186  memset(fatArea.getData(), 0, fatArea.getSize());
187  fatArea.getDataCast<uint32_t>()[0] =
188  0x0FFFFF00 + formatOptions.fatMediaType;
189  fatArea.getDataCast<uint32_t>()[1] = 0x0FFFFFFF;
190  fatArea.getDataCast<uint32_t>()[formatOptions.rootDirectoryCluster] =
191  0x0FFFFFFF;
192 
193  for (int i = 0; i < formatOptions.fatAreaCount; i++) {
194  result = iod->writeSector(fatArea.getDataCast<uint8_t>(),
195  offset + formatOptions.reservedSectorCount +
196  bpb.ebpbSectorsPerFAT32 * i,
197  bpb.ebpbSectorsPerFAT32);
198  if (result)
199  return result;
200  }
201 
202  uint8_t* emptycluster =
203  new uint8_t[iod->getDiskParameter().bytesPerSector *
204  formatOptions.sectorsPerCluster];
205  memset(emptycluster, 0, iod->getDiskParameter().bytesPerSector *
206  formatOptions.sectorsPerCluster);
207 
208  result = iod->writeSector(emptycluster,
209  offset + formatOptions.reservedSectorCount +
210  bpb.ebpbSectorsPerFAT32 * formatOptions.fatAreaCount +
211  formatOptions.sectorsPerCluster *
212  (formatOptions.rootDirectoryCluster - 2),
213  formatOptions.sectorsPerCluster);
214 
215  delete[] emptycluster;
216 
217  return OPENFSL_SUCCESS;
218 }
