16 size_t __getdigitcount(size_t n) {
17  size_t ret;
18  for (ret = 1;; ret++)
19  {
20  if (n >= 10) {
21  ret++;
22  n /= 10;
23  }
24  else
25  break;
26  }
27  return ret;
28 }

35  if (!std::regex_match(createFileInfo.fileName, checkNameValid))
36  return OPENFSL_ERROR_INVNAME;
37 
38  if (!ignoreReserved)
39  {
40  if (!isValidFileName(createFileInfo.fileName))

44  if (path != "") // Change directory if path is set

59

61  {
62  dotpos = createFileInfo.fileName.size();
63  if (dotpos > 8)
64  goto lfn;
65  else
66  goto sfn;
67  }
68 
69  if (dotpos > 8)
70  {
71  isLfn = true;
72  goto lfn;
73  }
74  else if (createFileInfo.fileName.size() > 13) {
75  lfn:
76  requiredEntrySize = (createFileInfo.fileName.size() / 13) + 2;
77  isLfn = true;
78  } else {
79  sfn:
80  requiredEntrySize = 1;
81  isLfn = false;
82  }
83 
84  // Check if file exists
85  std::vector<FileInfo> fileList;

89  currentCluster = tempCluster;

94  size_t dupEntrySize = 2;
95 
96  size_t lfnDuplicateFileCountInSfn = 0;
97 
98  for (auto & c: searchFilename) c = (char)toupper(c);
99 
100 recalc:
101  for (size_t i = 0; i < fileList.size(); i++)
102  {
103  std::string filename = fileList.at(i).fileName;
104  for (auto & c: filename) c = (char)toupper(c);
105 
106  if (filename.substr(0, 8 - dupEntrySize) == searchFilename.substr(0, 8 - dupEntrySize))
107  lfnDuplicateFileCountInSfn++;
108 
109  if (__getdigitcount(lfnDuplicateFileCountInSfn) != dupEntrySize - 1)
110  {
111  dupEntrySize++;
112  goto recalc;
113  }
114 
115  if (searchFilename == filename) {
116  currentPath = tempPath;
117  currentCluster = tempCluster;
118  return OPENFSL_ERROR_EXIST;
119  }
120  }
121 
122  // Find free entry
123  size_t entryPlace = 0; // SFN entry index(File entry last index) to create
124 
125  for (size_t i = 0; i < fileList.size(); i++)
126  {
127  size_t currentEntryStartIndex = fileList[i].fileEntryIndex - fileList[i].fileEntrySize + 1;
128  size_t blankEntryStartIndex;
129 
130  if (i == 0)
131  {
132  blankEntryStartIndex = 0;
133  }
134  else
135  {
136  blankEntryStartIndex = fileList[i - 1].fileEntryIndex + 1;
137  }
138 
139  size_t blankEntrySize = currentEntryStartIndex - blankEntryStartIndex;
140 
141  if (blankEntrySize >= requiredEntrySize)
142  {
143  entryPlace = blankEntrySize;
144  break;
145  }
146  else if (i == fileList.size() - 1)
147  {
148  entryPlace = fileList[i].fileEntryIndex + requiredEntrySize;
149  }
150  }
151 
152  LinkedCluster lcluster(*this, currentCluster);
153  size_t entryCount = lcluster.getClusterList()->size() * 16;
154 
155  if (entryPlace >= entryCount) // Add cluster to linked cluster if there's no place to create
156  {

158  if (result) {

161  return result;
162  }

164  }

170  fileEntry.fileSize = createFileInfo.fileSize;

174 
175  fileEntry.fileAccessDate = dateToWord(createFileInfo.fileAccessTime);
176 
177  fileEntry.fileModDate = dateToWord(createFileInfo.fileModTime);
178  fileEntry.fileModTime = timeToWord(createFileInfo.fileModTime);
179 
180  if (isLfn) { // If file name is LFN
181  // Convert LFN name to SFN
182  std::string filename_sfn = createFileInfo.fileName.substr(0, dotpos > 8 ? 8 : dotpos);
183  for (auto & c: filename_sfn) c = (char)toupper(c);
184  memset(fileEntry.fileName, ' ', 8);
185  memcpy(fileEntry.fileName, filename_sfn.c_str(), 8);
186 
187  std::string sfnDuplicateCountString = std::string("~") + std::to_string(lfnDuplicateFileCountInSfn + 1);
188  if (filename_sfn.size() < 6)
189  memcpy(fileEntry.fileName + filename_sfn.size(), sfnDuplicateCountString.c_str(),
190  sfnDuplicateCountString.size());
191  else
192  memcpy(fileEntry.fileName + 8 - sfnDuplicateCountString.size(), sfnDuplicateCountString.c_str(),
193  sfnDuplicateCountString.size());
194 
195  memset(fileEntry.fileExt, ' ', 3);
196 
197  if (dotpos != createFileInfo.fileName.size()) {
198  std::string fileext_sfn = createFileInfo.fileName.substr(dotpos + 1, 3);
199  for (auto & c: fileext_sfn) c = (char)toupper(c);
200 
201  memcpy(fileEntry.fileExt, fileext_sfn.c_str(), fileext_sfn.size());
202  }
203 
204  uint8_t checksum = chksumSFN(fileEntry.fileName);
205 
206  result = lcluster.write(&fileEntry, entryPlace * sizeof(FileEntry), sizeof(FileEntry));
207  if (result) {
208  currentPath = tempPath;
209  currentCluster = tempCluster;
210  return result;
211  }
212  requiredEntrySize--;
213 
214  // Write lfn entries
215  LFNEntry* lfnEntry = (LFNEntry*)&fileEntry;
216  lfnEntry->lfnSeqNum = 0;
217

219  for (entryPlace--;; entryPlace--)
220  {
221  memset(((uint8_t*)&fileEntry) + 1, 0, sizeof(LFNEntry) - 1);
222  lfnEntry->lfnSeqNum++;
223  if (requiredEntrySize < 2)
224  lfnEntry->lfnSeqNum += 0x40;
225  lfnEntry->lfnAttr = FileAttribute::LFN;
226  lfnEntry->lfnCheckSum = checksum;
227 
228  if (wideFileName.size() >= 5) {
229  memcpy(lfnEntry->lfnFileName1, wideFileName.c_str(), 10);
230  wideFileName = wideFileName.substr(5);
231  }
232  else if (wideFileName.size() > 0) {
233  memset(lfnEntry->lfnFileName1, 0xFF, 10);
234  memcpy(lfnEntry->lfnFileName1, wideFileName.c_str(), wideFileName.length() * sizeof(char16_t));
235  wideFileName.clear();
236  }
237  else {
238  memset(lfnEntry->lfnFileName1, 0xFF, 10);
239  }
240 
241  if (wideFileName.size() >= 6) {
242  memcpy(lfnEntry->lfnFileName2, wideFileName.c_str(), 12);
243  wideFileName = wideFileName.substr(6);
244  }
245  else if (wideFileName.size() > 0) {
246  memset(lfnEntry->lfnFileName2, 0xFF, 12);
247  memcpy(lfnEntry->lfnFileName2, wideFileName.c_str(), wideFileName.length() * sizeof(char16_t));
248  wideFileName.clear();
249  }
250  else {
251  memset(lfnEntry->lfnFileName2, 0xFF, 12);
252  }
253

255  memcpy(lfnEntry->lfnFileName3, wideFileName.c_str(), 4);
256  wideFileName = wideFileName.substr(2);
257  }
258  else if (wideFileName.size() > 0) {
259  memset(lfnEntry->lfnFileName3, 0xFF, 4);
260  memcpy(lfnEntry->lfnFileName3, wideFileName.c_str(), wideFileName.length() * sizeof(char16_t));
261  wideFileName.clear();
262  }
263  else {
264  memset(lfnEntry->lfnFileName3, 0xFF, 4);
265  }
266 
267  result = lcluster.write(&fileEntry, entryPlace * sizeof(FileEntry), sizeof(FileEntry));
268  if (result) {
269  currentPath = tempPath;
270  currentCluster = tempCluster;
271  return result;
272  }
273 
274  if (requiredEntrySize < 2)
275  break;
276 
277  requiredEntrySize--;
278  }
279  }
280  else // If file name is SFN
281  {
282  for (auto & c: createFileInfo.fileName) c = (char)toupper(c);
283  std::string filename_str = createFileInfo.fileName.substr(0, dotpos);
284  std::string fileext_str = "";
285 
286  if (createFileInfo.fileName == ".")
287  {
288  memset(fileEntry.fileName, ' ', 8);
289  memset(fileEntry.fileExt, ' ', 3);
290  fileEntry.fileName[0] = '.';
291  }
292  else if (createFileInfo.fileName == "..") {
293  memset(fileEntry.fileName, ' ', 8);
294  memset(fileEntry.fileExt, ' ', 3);
295  fileEntry.fileName[0] = '.';
296  fileEntry.fileName[1] = '.';
297  }
298  else {
299  memcpy(fileEntry.fileName, filename_str.c_str(), dotpos);
300  memset(fileEntry.fileName + filename_str.size(), ' ', 8 - filename_str.size());
301 
302  if (dotpos != createFileInfo.fileName.size()) {
303  fileext_str = createFileInfo.fileName.substr(dotpos + 1);
304  }
305  memcpy(fileEntry.fileExt, fileext_str.c_str(), fileext_str.size());
306  memset(fileEntry.fileExt + fileext_str.size(), ' ', 3 - fileext_str.size());
307  }
308 
309  result = lcluster.write(&fileEntry, entryPlace * sizeof(FileEntry), sizeof(FileEntry));
310  if (result) {
311  currentPath = tempPath;
312  currentCluster = tempCluster;
313  return result;
314  }
315  }
316 
317  result = lcluster.flush();
318  if (result) {
319  currentPath = tempPath;
320  currentCluster = tempCluster;
321  return result;
322  }
323

325  currentCluster = tempCluster;
326 
327  return 0;
