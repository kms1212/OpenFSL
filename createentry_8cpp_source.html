14 size_t __getdigitcount(size_t n) {
15  size_t ret;
16  for (ret = 1;; ret++) {
17  if (n >= 10) {
18  ret++;
19  n /= 10;
20  } else {
21  break;
22  }
23  }
24  return ret;
25 }

27 error_t openfsl::FAT32::__createEntry(
28  const std::string path, const FileInfo createFileInfo) {
29  std::string tempPath = currentPath;
30  cluster_t tempCluster = currentCluster;
31 
32  if (createFileInfo.fileSize > std::numeric_limits<uint32_t>::max())
33  return OPENFSL_ERROR_TOO_LARGE_FILE_SIZE;
34 
35  if (path != "") { // Change directory if path is set
36  int errcode = __chdir(path);
37  if (errcode) {
38  currentPath = tempPath;
39  currentCluster = tempCluster;
40  return errcode;
41  }
42  }
43 
44  // Validate file name and get required entry count
45  size_t requiredEntrySize =
46  getEntryCount(createFileInfo.fileName, true);
47  size_t dotpos = createFileInfo.fileName.find_last_of('.');
48 
49  if (requiredEntrySize == 0) {
50  return OPENFSL_ERROR_INVALID_FILE_NAME;
51  }
52  if (dotpos == std::string::npos) {
53  dotpos = createFileInfo.fileName.size();
54  }
55 
56  // Check if file exists
57  std::vector<FileInfo> fileList;
58  error_t result = __listDir(&fileList, FileAttribute::All);
59  if (result) {
60  currentPath = tempPath;
61  currentCluster = tempCluster;
62  return result;
63  }
64 
65  std::string searchFilename = createFileInfo.fileName;
66  size_t dupEntrySize = 2;
67 
68  size_t lfnDuplicateFileCountInSfn = 0;
69 
70  for (auto & c : searchFilename) c = static_cast<char>(toupper(c));
71 
72 recalc:
73  for (size_t i = 0; i < fileList.size(); i++) {
74  std::string filename = fileList.at(i).fileName;
75  for (auto & c : filename) c = static_cast<char>(toupper(c));
76 
77  if (filename.substr(0, 8 - dupEntrySize) ==
78  searchFilename.substr(0, 8 - dupEntrySize))
79  lfnDuplicateFileCountInSfn++;
80 
81  if (__getdigitcount(lfnDuplicateFileCountInSfn) != dupEntrySize - 1) {
82  dupEntrySize++;
83  goto recalc;
84  }
85 
86  if (searchFilename == filename) {
87  currentPath = tempPath;
88  currentCluster = tempCluster;
89  return OPENFSL_ERROR_FILE_OR_DIR_EXISTS;
90  }
91  }
92 
93  // Find free entry
94  LinkedCluster lcluster(this, currentCluster);
95  size_t entryPlace = 0;
96  size_t blankPlaceCount = 0;
97  size_t totalEntryCount = lcluster.getClusterList()->size() * 16;
98 
99  for (size_t i = 0; i < totalEntryCount + 1; i++) {
100  if (i == totalEntryCount) {
101  result = lcluster.addCluster(1);
102  if (result) {
103  currentPath = tempPath;
104  currentCluster = tempCluster;
105  return result;
106  }
107  totalEntryCount = lcluster.getClusterList()->size() * 16;
108  }
109 
110  uint8_t deletionMark;
111  result = lcluster.read(&deletionMark, i * 32, 1);
112  if (result) {
113  currentPath = tempPath;
114  currentCluster = tempCluster;
115  return result;
116  }
117 
118  if ((deletionMark == 0xE5) || (deletionMark == 0x00))
119  blankPlaceCount++;
120  else
121  blankPlaceCount = 0;
122 
123  if (blankPlaceCount >= requiredEntrySize) {
124  entryPlace = i;
125  break;
126  }
127  }
128 
129  // Create entry
130  FileEntry fileEntry;
131  fileEntry.fileAttr = createFileInfo.fileAttr;
132  fileEntry.fileLocationHigh = (uint16_t)(createFileInfo.fileLocation >> 16);
133  fileEntry.fileLocationLow = (uint16_t)createFileInfo.fileLocation;
134  fileEntry.fileSize = (uint32_t)createFileInfo.fileSize;
135 
136  fileEntry.fileCreateDate = __dateToWord(createFileInfo.fileCreateTime);
137  fileEntry.fileCreateTime = __timeToWord(createFileInfo.fileCreateTime);
138 
139  fileEntry.fileAccessDate = __dateToWord(createFileInfo.fileAccessTime);
140 
141  fileEntry.fileModDate = __dateToWord(createFileInfo.fileModTime);
142  fileEntry.fileModTime = __timeToWord(createFileInfo.fileModTime);
143 
144  if (requiredEntrySize > 1) { // If file name is LFN
145  // Initialize SFN entry to whitespace
146  memset(fileEntry.fileName, ' ', 11);
147 
148  std::u16string filename16 = __u8ToUCS2(createFileInfo.fileName);
149 
150  // Convert LFN name to SFN
151 
152  // Copy first 8 chars from filename to entry and replace every
153  // non-ASCII character to underscore
154  std::string filename_sfn;
155  for (auto c : filename16.substr(0, dotpos > 8 ? 8 : dotpos))
156  filename_sfn
157  .push_back((c < 0x80) && (c != '.') ?
158  static_cast<char>(toupper(c)) : '_');
159  memcpy(fileEntry.fileName, filename_sfn.c_str(), filename_sfn.size());
160 
161  // Append duplicate count string to filename
162  if (dotpos > 8) {
163  std::string sfnDuplicateCountString = std::string("~") +
164  std::to_string(lfnDuplicateFileCountInSfn + 1);
165  memcpy(fileEntry.fileName + 8 - sfnDuplicateCountString.size(),
166  sfnDuplicateCountString.c_str(),
167  sfnDuplicateCountString.size());
168  }
169 
170  // Copy first 3 characters from file extension to entry and replace
171  // every non-ASCII character to underscore
172  if (dotpos != filename16.size()) {
173  std::string fileext_sfn;
174  size_t extsize = filename16.size() - dotpos;
175  for (auto c :
176  filename16.substr(dotpos + 1, extsize > 3 ? 3 : extsize))
177  fileext_sfn
178  .push_back(c < 0x80 ? static_cast<char>(toupper(c)) : '_');
179 
180  memcpy(fileEntry.fileExt, fileext_sfn.c_str(), fileext_sfn.size());
181  }
182 
183  uint8_t checksum = __chksumSFN(fileEntry.fileName);
184 
185  // Write SFN entry
186  result = lcluster.write(&fileEntry,
187  entryPlace * sizeof(FileEntry), sizeof(FileEntry));
188  if (result) {
189  currentPath = tempPath;
190  currentCluster = tempCluster;
191  return result;
192  }
193  requiredEntrySize--;
194 
195  // Create lfn entries
196  LFNEntry* lfnEntry = reinterpret_cast<LFNEntry*>(&fileEntry);
197  lfnEntry->lfnSeqNum = 0;
198 
199  for (entryPlace--;; entryPlace--) {
200  memset((reinterpret_cast<uint8_t*>(&fileEntry)) + 1, 0,
201  sizeof(LFNEntry) - 1);
202  lfnEntry->lfnSeqNum++;
203  if (requiredEntrySize < 2)
204  lfnEntry->lfnSeqNum += 0x40;
205  lfnEntry->lfnAttr = FileAttribute::LFN;
206  lfnEntry->lfnCheckSum = checksum;
207 
208  if (filename16.size() >= 5) {
209  memcpy(lfnEntry->lfnFileName1, filename16.c_str(), 10);
210  filename16 = filename16.substr(5);
211  } else if (filename16.size() > 0) {
212  memset(lfnEntry->lfnFileName1, 0xFF, 10);
213  memcpy(lfnEntry->lfnFileName1, filename16.c_str(),
214  filename16.length() * sizeof(char16_t));
215  filename16.clear();
216  } else {
217  memset(lfnEntry->lfnFileName1, 0xFF, 10);
218  }
219 
220  if (filename16.size() >= 6) {
221  memcpy(lfnEntry->lfnFileName2, filename16.c_str(), 12);
222  filename16 = filename16.substr(6);
223  } else if (filename16.size() > 0) {
224  memset(lfnEntry->lfnFileName2, 0xFF, 12);
225  memcpy(lfnEntry->lfnFileName2, filename16.c_str(),
226  filename16.length() * sizeof(char16_t));
227  filename16.clear();
228  } else {
229  memset(lfnEntry->lfnFileName2, 0xFF, 12);
230  }
231 
232  if (filename16.size() >= 2) {
233  memcpy(lfnEntry->lfnFileName3, filename16.c_str(), 4);
234  filename16 = filename16.substr(2);
235  } else if (filename16.size() > 0) {
236  memset(lfnEntry->lfnFileName3, 0xFF, 4);
237  memcpy(lfnEntry->lfnFileName3, filename16.c_str(),
238  filename16.length() * sizeof(char16_t));
239  filename16.clear();
240  } else {
241  memset(lfnEntry->lfnFileName3, 0xFF, 4);
242  }
243 
244  // Write current LFN entry
245  result = lcluster.write(&fileEntry,
246  entryPlace * sizeof(FileEntry), sizeof(FileEntry));
247  if (result) {
248  currentPath = tempPath;
249  currentCluster = tempCluster;
250  return result;
251  }
252 
253  if (requiredEntrySize < 2)
254  break;
255 
256  requiredEntrySize--;
257  }
258  } else { // If file name is SFN
259  std::string createFilename = createFileInfo.fileName;
260  for (auto & c : createFilename) c = static_cast<char>(toupper(c));
261  std::string filename_str = createFilename.substr(0, dotpos);
262  std::string fileext_str = "";
263 
264  if (createFilename == ".") {
265  memset(fileEntry.fileName, ' ', 8);
266  memset(fileEntry.fileExt, ' ', 3);
267  fileEntry.fileName[0] = '.';
268  } else if (createFilename == "..") {
269  memset(fileEntry.fileName, ' ', 8);
270  memset(fileEntry.fileExt, ' ', 3);
271  fileEntry.fileName[0] = '.';
272  fileEntry.fileName[1] = '.';
273  } else {
274  memcpy(fileEntry.fileName, filename_str.c_str(), dotpos);
275  memset(fileEntry.fileName + filename_str.size(),
276  ' ', 8 - filename_str.size());
277 
278  if (dotpos != createFilename.size()) {
279  fileext_str = createFilename.substr(dotpos + 1);
280  }
281  memcpy(fileEntry.fileExt, fileext_str.c_str(), fileext_str.size());
282  memset(fileEntry.fileExt + fileext_str.size(),
283  ' ', 3 - fileext_str.size());
284  }
285 
286  result = lcluster.write(&fileEntry,
287  entryPlace * sizeof(FileEntry), sizeof(FileEntry));
288  if (result) {
289  currentPath = tempPath;
290  currentCluster = tempCluster;
291  return result;
292  }
293  }
294 
295  result = lcluster.flush();
296  if (result) {
297  currentPath = tempPath;
298  currentCluster = tempCluster;
299  return result;
300  }
301 
302  currentPath = tempPath;
303  currentCluster = tempCluster;
304 
305  return 0;
306 }
