16 size_t __getdigitcount(size_t n) {
17  size_t ret;
18  for (ret = 1;; ret++)
19  {
20  if (n >= 10) {
21  ret++;
22  n /= 10;
23  }
24  else
25  break;
26  }
27  return ret;
28 }

30 error_t FAT32::createEntry(std::string path, FileInfo createFileInfo, bool ignoreReserved) {

32  cluster_t tempCluster = currentCluster;
33 
34  std::regex checkNameValid("^[^<>:\"\\/\\\\|?\\*]*\\s*$", std::regex::icase);
35  if (!std::regex_match(createFileInfo.fileName, checkNameValid))
36  return OPENFSL_ERROR_INVNAME;
37 
38  if (!ignoreReserved)
39  {
40  for (std::string regexstr : reservedName)
41  {
42  checkNameValid.assign(regexstr, std::regex::icase);
43  if (!std::regex_match(createFileInfo.fileName, checkNameValid))
44  return OPENFSL_ERROR_INVNAME;
45  }
46  }
47 
48  if (path != "") // Change directory if path is set
49  {
50  int errcode = chdir(path);
51  if (errcode)
52  {
53  currentPath = tempPath;
54  currentCluster = tempCluster;

56  }
57  }
58 
59  // Check if string is lfn or sfn
60  bool isLfn = false;
61  size_t requiredEntrySize = 0;
62  size_t dotpos = createFileInfo.fileName.find_last_of('.');
63 
64  if (dotpos == std::string::npos)
65  {
66  dotpos = createFileInfo.fileName.size();
67  if (dotpos > 8)
68  goto lfn;
69  else
70  goto sfn;
71  }
72 
73  if (dotpos > 8)
74  {
75  isLfn = true;
76  goto lfn;
77  }
78  else if (createFileInfo.fileName.size() > 13) {
79  lfn:
80  requiredEntrySize = (createFileInfo.fileName.size() / 13) + 2;
81  isLfn = true;
82  } else {
83  sfn:
84  requiredEntrySize = 1;
85  isLfn = false;
86  }
87 
88  // Check if file exists
89  std::vector<FileInfo> fileList;
90  error_t result = listDir(fileList, FileAttribute::All);
91  if (result) {
92  currentPath = tempPath;
93  currentCluster = tempCluster;
94  return result;
95  }
96 
97  std::string searchFilename = createFileInfo.fileName;
98  size_t dupEntrySize = 2;
99 
100  size_t lfnDuplicateFileCountInSfn = 0;
101 
102  for (auto & c: searchFilename) c = (char)toupper(c);
103 
104 recalc:
105  for (size_t i = 0; i < fileList.size(); i++)
106  {
107  std::string filename = fileList.at(i).fileName;
108  for (auto & c: filename) c = (char)toupper(c);
109 
110  if (filename.substr(0, 8 - dupEntrySize) == searchFilename.substr(0, 8 - dupEntrySize))
111  lfnDuplicateFileCountInSfn++;
112 
113  if (__getdigitcount(lfnDuplicateFileCountInSfn) != dupEntrySize - 1)
114  {
115  dupEntrySize++;
116  goto recalc;
117  }
118 
119  if (searchFilename == filename) {
120  currentPath = tempPath;
121  currentCluster = tempCluster;
122  return OPENFSL_ERROR_EXIST;
123  }
124  }
125 
126  // Find free entry
127  size_t entryPlace = 0; // SFN entry index(File entry last index) to create
128 
129  for (size_t i = 0; i < fileList.size(); i++)
130  {
131  size_t currentEntryStartIndex = fileList[i].fileEntryIndex - fileList[i].fileEntrySize + 1;
132  size_t blankEntryStartIndex;
133 
134  if (i == 0)
135  {
136  blankEntryStartIndex = 0;
137  }
138  else
139  {
140  blankEntryStartIndex = fileList[i - 1].fileEntryIndex + 1;
141  }
142 
143  size_t blankEntrySize = currentEntryStartIndex - blankEntryStartIndex;
144 
145  if (blankEntrySize >= requiredEntrySize)
146  {
147  entryPlace = blankEntrySize;
148  break;
149  }
150  else if (i == fileList.size() - 1)
151  {
152  entryPlace = fileList[i].fileEntryIndex + requiredEntrySize;
153  }
154  }
155 
156  LinkedCluster lcluster(*this, currentCluster);
157  size_t entryCount = lcluster.getClusterList()->size() * 16;
158 
159  if (entryPlace > entryCount) // Add cluster to linked cluster if there's no place to create
160  {
161  result = lcluster.addCluster(1);
162  if (result) {
163  currentPath = tempPath;
164  currentCluster = tempCluster;
165  return result;
166  }
167  entryCount = lcluster.getClusterList()->size() * 16;
168  }
169 
170  FileEntry fileEntry;
171  fileEntry.fileAttr = createFileInfo.fileAttr;
172  fileEntry.fileLocationHigh = (uint16_t)(createFileInfo.fileLocation >> 16);
173  fileEntry.fileLocationLow = (uint16_t)createFileInfo.fileLocation;
174  fileEntry.fileSize = createFileInfo.fileSize;
175 
176  fileEntry.fileCreateDate = dateToWord(createFileInfo.fileCreateTime);
177  fileEntry.fileCreateTime = timeToWord(createFileInfo.fileCreateTime);
178 
179  fileEntry.fileAccessDate = dateToWord(createFileInfo.fileAccessTime);
180 
181  fileEntry.fileModDate = dateToWord(createFileInfo.fileModTime);
182  fileEntry.fileModTime = timeToWord(createFileInfo.fileModTime);
183 
184  if (isLfn) { // If file name is LFN
185  // Convert LFN name to SFN
186  std::string filename_sfn = createFileInfo.fileName.substr(0, dotpos > 8 ? 8 : dotpos);
187  for (auto & c: filename_sfn) c = (char)toupper(c);
188  memset(fileEntry.fileName, ' ', 8);
189  memcpy(fileEntry.fileName, filename_sfn.c_str(), 8);
190 
191  std::string sfnDuplicateCountString = std::string("~") + std::to_string(lfnDuplicateFileCountInSfn + 1);
192  if (filename_sfn.size() < 6)
193  memcpy(fileEntry.fileName + filename_sfn.size(), sfnDuplicateCountString.c_str(),
194  sfnDuplicateCountString.size());
195  else
196  memcpy(fileEntry.fileName + 8 - sfnDuplicateCountString.size(), sfnDuplicateCountString.c_str(),
197  sfnDuplicateCountString.size());
198 
199  memset(fileEntry.fileExt, ' ', 3);
200 
201  if (dotpos != createFileInfo.fileName.size()) {
202  std::string fileext_sfn = createFileInfo.fileName.substr(dotpos + 1, 3);
203  for (auto & c: fileext_sfn) c = (char)toupper(c);
204 
205  memcpy(fileEntry.fileExt, fileext_sfn.c_str(), fileext_sfn.size());
206  }
207 
208  uint8_t checksum = chksumSFN(fileEntry.fileName);
209 
210  result = lcluster.write(&fileEntry, entryPlace * sizeof(FileEntry), sizeof(FileEntry));
211  if (result) {
212  currentPath = tempPath;
213  currentCluster = tempCluster;
214  return result;
215  }
216  requiredEntrySize--;
217 
218  // Write lfn entries
219  LFNEntry* lfnEntry = (LFNEntry*)&fileEntry;
220  lfnEntry->lfnSeqNum = 0;
221 
222  std::u16string wideFileName = u8ToUCS2(createFileInfo.fileName);
223  for (entryPlace--;; entryPlace--)
224  {
225  memset(((uint8_t*)&fileEntry) + 1, 0, sizeof(LFNEntry) - 1);
226  lfnEntry->lfnSeqNum++;
227  if (requiredEntrySize < 2)
228  lfnEntry->lfnSeqNum += 0x40;
229  lfnEntry->lfnAttr = FileAttribute::LFN;
230  lfnEntry->lfnCheckSum = checksum;
231 
232  if (wideFileName.size() >= 5) {
233  memcpy(lfnEntry->lfnFileName1, wideFileName.c_str(), 10);
234  wideFileName = wideFileName.substr(5);
235  }
236  else if (wideFileName.size() > 0) {
237  memset(lfnEntry->lfnFileName1, 0xFF, 10);
238  memcpy(lfnEntry->lfnFileName1, wideFileName.c_str(), wideFileName.length() * sizeof(char16_t));
239  wideFileName.clear();
240  }
241  else {
242  memset(lfnEntry->lfnFileName1, 0xFF, 10);
243  }
244 
245  if (wideFileName.size() >= 6) {
246  memcpy(lfnEntry->lfnFileName2, wideFileName.c_str(), 12);
247  wideFileName = wideFileName.substr(6);
248  }
249  else if (wideFileName.size() > 0) {
250  memset(lfnEntry->lfnFileName2, 0xFF, 12);
251  memcpy(lfnEntry->lfnFileName2, wideFileName.c_str(), wideFileName.length() * sizeof(char16_t));
252  wideFileName.clear();
253  }
254  else {
255  memset(lfnEntry->lfnFileName2, 0xFF, 12);
256  }
257 
258  if (wideFileName.size() >= 2) {
259  memcpy(lfnEntry->lfnFileName3, wideFileName.c_str(), 4);
260  wideFileName = wideFileName.substr(2);
261  }
262  else if (wideFileName.size() > 0) {
263  memset(lfnEntry->lfnFileName3, 0xFF, 4);
264  memcpy(lfnEntry->lfnFileName3, wideFileName.c_str(), wideFileName.length() * sizeof(char16_t));
265  wideFileName.clear();
266  }
267  else {
268  memset(lfnEntry->lfnFileName3, 0xFF, 4);
269  }
270 
271  result = lcluster.write(&fileEntry, entryPlace * sizeof(FileEntry), sizeof(FileEntry));
272  if (result) {
273  currentPath = tempPath;
274  currentCluster = tempCluster;
275  return result;
276  }
277 
278  if (requiredEntrySize < 2)
279  break;
280 
281  requiredEntrySize--;
282  }
283  }
284  else // If file name is SFN
285  {
286  for (auto & c: createFileInfo.fileName) c = (char)toupper(c);
287  std::string filename_str = createFileInfo.fileName.substr(0, dotpos);
288  std::string fileext_str = "";
289 
290  if (createFileInfo.fileName == ".")
291  {
292  memset(fileEntry.fileName, ' ', 8);
293  memset(fileEntry.fileExt, ' ', 3);
294  fileEntry.fileName[0] = '.';
295  }
296  else if (createFileInfo.fileName == "..") {
297  memset(fileEntry.fileName, ' ', 8);
298  memset(fileEntry.fileExt, ' ', 3);
299  fileEntry.fileName[0] = '.';
300  fileEntry.fileName[1] = '.';
301  }
302  else {
303  memcpy(fileEntry.fileName, filename_str.c_str(), dotpos);
304  memset(fileEntry.fileName + filename_str.size(), ' ', 8 - filename_str.size());
305 
306  if (dotpos != createFileInfo.fileName.size()) {
307  fileext_str = createFileInfo.fileName.substr(dotpos + 1);
308  }
309  memcpy(fileEntry.fileExt, fileext_str.c_str(), fileext_str.size());
310  memset(fileEntry.fileExt + fileext_str.size(), ' ', 3 - fileext_str.size());
311  }
312 
313  result = lcluster.write(&fileEntry, entryPlace * sizeof(FileEntry), sizeof(FileEntry));
314  if (result) {
315  currentPath = tempPath;
316  currentCluster = tempCluster;
317  return result;
318  }
319  }
320 
321  result = lcluster.flush();
322  if (result) {
323  currentPath = tempPath;
324  currentCluster = tempCluster;
325  return result;
326  }
327 
328  currentPath = tempPath;
329  currentCluster = tempCluster;
330 
331  return 0;
332 }
