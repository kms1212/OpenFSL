14 int openFSL::detectDiskStructure(DiskStructure& buf, DiskDevice& dd) {
15  Sector checkSector(1, dd.getDiskParameter().bytesPerSector);
16  error_t result = dd.readDisk(checkSector.getData(), 0, 1);
17  if (result)
18  return result;
19 
20  buf.partTable = PartitionTableType::pt_None;
21  buf.partList.clear();
22 
23  fsStructure::COMMAA55h* tailSigChk = (fsStructure::COMMAA55h*)checkSector.getData();
24 
25  if (tailSigChk->tailSig == 0xAA55) {
26  fsStructure::FAT1216BPB* fat1216Chk = (fsStructure::FAT1216BPB*)checkSector.getData();
27  if (memcmp(fat1216Chk->fatFsType, "FAT12 ", 8) == 0)
28  {
29  buf.partList.push_back(FileSystemType::fs_FAT12);
30  return 0;
31  }
32  else if (memcmp(fat1216Chk->fatFsType, "FAT16 ", 8) == 0)
33  {
34  buf.partList.push_back(FileSystemType::fs_FAT16);
35  return 0;
36  }
37 
38  fsStructure::FAT32BPB* fat32Chk = (fsStructure::FAT32BPB*)checkSector.getData();
39  if (memcmp(fat32Chk->fatFsType, "FAT32 ", 8) == 0)
40  {
41  buf.partList.push_back(FileSystemType::fs_FAT32);
42  return 0;
43  }
44 
45  fsStructure::NTFSBPB* ntfsChk = (fsStructure::NTFSBPB*)checkSector.getData();
46  if (memcmp(ntfsChk->fsType, "NTFS ", 8) == 0)
47  {
48  buf.partList.push_back(FileSystemType::fs_NTFS);
49  return 0;
50  }
51 
52  buf.partTable = PartitionTableType::pt_MBR;

55

57  for (size_t i = 0; i < fsListInMBR.size(); i++)
58  {
59  buf.partList.push_back(fsListInMBR[i]);
60  }
61  }
62 
63  return 0;
64 }
