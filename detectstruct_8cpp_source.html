15 error_t openfsl::detectDiskStructure(DiskStructure* buf, BlockDevice* bd) {
16  Sector checkSector(1, bd->getDiskParameter().bytesPerSector);
17  error_t result = bd->readSector(checkSector.getData(), 0, 1);
18  if (result)
19  return result;
20 
21  buf->partTable = PartitionTableType::None;
22  buf->partList.clear();
23 
24  fsStructure::COMMAA55h* tailSigChk =
25  checkSector.getDataCast<fsStructure::COMMAA55h>();
26 
27  if (tailSigChk->tailSig == 0xAA55) {
28  fsStructure::FAT1216BPB* fat1216Chk =
29  checkSector.getDataCast<fsStructure::FAT1216BPB>();
30  if (memcmp(fat1216Chk->fatFsType, "FAT12 ", 8) == 0) {
31  buf->partList.push_back(FileSystemType::FAT12);
32  return 0;
33  } else if (memcmp(fat1216Chk->fatFsType, "FAT16 ", 8) == 0) {
34  buf->partList.push_back(FileSystemType::FAT16);
35  return 0;
36  }
37

44

48  buf->partList.push_back(FileSystemType::NTFS);

51

54  if (result)

67  // fsList = gpt.getPartitionInfo();
68  } else {

70  }
71 
72  for (size_t i = 0; i < fsList.size(); i++) {

74  }
75  }

78 }
