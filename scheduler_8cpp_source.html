14 int Scheduler::runTask(std::function<int()> task, int* returnVal, double* execTime) {
15  taskQueueLock.lock();
16 
17  Task taskTemp; // Copy values to temp variable
18  taskTemp.key = currentLastKey++;
19  taskTemp.task = task;
20  taskQueue.push(taskTemp);
21 
22  taskQueueLock.unlock();
23 
24  for (;;) { // Infinite loop
25  taskQueueLock.lock();
26  if (taskQueue.empty()) { // Return error if task queue empty while waiting task to run. (Aborted)
27  taskQueueLock.unlock();
28  return OPENFSL_ERROR_SCHABRT;
29  }
30  else {
31  Task currentTask = taskQueue.front();
32 
33  if (currentTask.key == taskTemp.key) { // If key value of front of queue is same with temp, run task.
34  clock_t start = clock();
35 
36  if (returnVal == nullptr)
37  currentTask.task();
38  else
39  *returnVal = currentTask.task();
40 
41  if (execTime != nullptr) {
42  clock_t end = clock();
43  *execTime = (double(end) - double(start)) / double(CLOCKS_PER_SEC); // Calculate execution time
44  }
45 
46  taskQueue.pop();
47 
48  break;
49  }
50  }
51  taskQueueLock.unlock();
52  }
53 
54  taskQueueLock.unlock();
55  return 0;
56 }
