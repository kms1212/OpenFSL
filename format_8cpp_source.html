16 FormatOptions FS_FAT32::getDefaultFormatOptions(lba48_t fsSize) {
17  FormatOptions ret;
18  ret.volumeLabel = "NO NAME";
19 
20  std::chrono::system_clock::time_point tpoint = std::chrono::system_clock::now();
21  time_t ttenth = std::chrono::duration_cast<std::chrono::seconds>(tpoint.time_since_epoch()).count() / 10;
22  ttenth %= 100;
23  time_t t = std::chrono::system_clock::to_time_t(tpoint);
24  std::tm currentTime = fsl_localtime(t);
25 
26  uint16_t tmp1 = (((uint8_t)currentTime.tm_mon + 1) << 8) + (uint8_t)currentTime.tm_mday;
27  uint16_t tmp2 = ((uint8_t)currentTime.tm_sec << 8) + (uint8_t)ttenth;
28  ret.volumeSerial = (tmp1 + tmp2) << 16;
29 
30  tmp1 = ((uint8_t)currentTime.tm_hour << 8) + (uint8_t)currentTime.tm_min;
31  tmp2 = (uint16_t)currentTime.tm_year + 1900;
32  ret.volumeSerial += tmp1 + tmp2;
33 
34  ret.biosDriveNum = 0x80;
35 
36  if (fsSize < 532480)
37  ret.sectorsPerCluster = 1;
38  else if (fsSize < 16777216)
39  ret.sectorsPerCluster = 8;
40  else if (fsSize < 33554432)
41  ret.sectorsPerCluster = 16;
42  else if (fsSize < 67108864)
43  ret.sectorsPerCluster = 32;
44  else
45  ret.sectorsPerCluster = 64;
46 
47  ret.backupBSLocation = 6;
48  ret.fsinfoLocation = 1;
49  ret.reservedSectorCount = 32;
50 
51  ret.fatAreaCount = 2;
52  ret.fatMediaType = 0xF8;
53 
54  return ret;
55 }

57 void __hexdump(const uint8_t* p, size_t offset, size_t len)
58 {
59  std::ios init(NULL);
60  init.copyfmt(std::cout);
61 
62  size_t address = 0;
63  size_t row = 0;
64 
65  std::cout << std::hex << std::setfill('0');
66  while (1) {
67  if (address >= len) break;
68  size_t nread = ((len - address) > 16) ? 16 : (len - address);
69 
70  // Show the address
71  std::cout << std::right << std::setw(8) << address + offset;
72 
73  // Show the hex codes
74  for (size_t i = 0; i < 16; i++)
75  {
76  if (i % 8 == 0) std::cout << ' ';
77  if (i < nread)
78  std::cout << ' ' << std::setw(2) << (int)p[16 * row + i + offset];
79  else
80  std::cout << " ";
81  }
82 
83  // Show printable characters
84  std::cout << " ";
85  for (size_t i = 0; i < nread; i++)
86  {
87  uint8_t ch = p[16 * row + i + offset];
88  if (ch < 32 || ch > 126) std::cout << '.';
89  else std::cout << ch;
90  }
91 
92  std::cout << "\n";
93  address += 16;
94  row++;
95  }
96 
97  std::cout.flush();
98  std::cout.copyfmt(init);
99 }

101 error_t FS_FAT32::format(lba48_t offset, lba48_t size, FormatOptions formatOptions) {
102  BPBSector bpb = { 0 };
103 
104  // Set signatures
105  bpb.bpbVBRSignature = 0xAA55;
106  bpb.ebpbExtendedBootSign = 0x29;
107 
108  // Copy VBR boot codes
109  bpb.bpbJumpCode[0] = 0xEB;
110  bpb.bpbJumpCode[1] = 0x58;
111  bpb.bpbJumpCode[2] = 0x90;
112 
113  memcpy(bpb.bpbBootCode, _blob_vbrboot_bin_data, _blob_vbrboot_bin_len);
114 
115  // Copy volume OEM ID, Label, Volume ID, FS Type
116  if (formatOptions.volumeLabel.length() > 11)
117  {
118  return OPENFSL_ERROR_VLABLONG; // Volume label is too long
119  }
120 
121  memcpy(bpb.bpbOEMName, "OPENFSL ", 8);
122 
123  memset(bpb.ebpbVolumeLabel, ' ', 11);
124  memcpy(bpb.ebpbVolumeLabel, formatOptions.volumeLabel.c_str(), formatOptions.volumeLabel.length());
125 
126  bpb.ebpbVolumeSerial = (uint16_t)formatOptions.volumeSerial;
127 
128  memcpy(bpb.ebpbFSType, "FAT32 ", 8);
129 
130  // Disk informations
131  if (formatOptions.biosDriveNum < 80)
132  {
133  return OPENFSL_ERROR_NOFLOPPY;
134  }
135 
136  bpb.bpbMedia = 0xF8;
137  bpb.ebpbPhysicalDriveNum = formatOptions.biosDriveNum;
138  bpb.bpbBytesPerSector = getDiskDevice()->getDiskParameter().bytesPerSector;
139  bpb.bpbSectorsPerTrack = getDiskDevice()->getDiskParameter().sectorPerTrack;
140  bpb.bpbDiskHeads = getDiskDevice()->getDiskParameter().headPerCylinder;
141 
142  // Cluster, sector informations
143  bpb.bpbHiddenSectors = (uint32_t)offset;
144  bpb.bpbTotalSector32 = (uint32_t)size;
145 
146  bpb.bpbSectorPerCluster = formatOptions.sectorsPerCluster;
147  bpb.bpbReservedSectors = formatOptions.reservedSectorCount;
148 
149  if ((formatOptions.backupBSLocation >= formatOptions.reservedSectorCount) ||
150  (formatOptions.fsinfoLocation >= formatOptions.reservedSectorCount))
151  {
152  return OPENFSL_ERROR_INVFMTOPT;
153  }
154 
155  bpb.ebpbBSBackupSector = formatOptions.backupBSLocation;
156  bpb.ebpbFSInfoSector = formatOptions.fsinfoLocation;
157 
158  // FAT Informations
159  cluster_t tmpval1 = size - formatOptions.reservedSectorCount;
160  cluster_t tmpval2 = 128 * formatOptions.sectorsPerCluster + formatOptions.fatAreaCount;
161 
162  bpb.bpbFATCount = formatOptions.fatAreaCount;
163  bpb.ebpbSectorsPerFAT32 = (tmpval1 + tmpval2 - 1) / tmpval2;
164 
165 
166  __hexdump((uint8_t*)&bpb, 0, 512);
167 
168  return OPENFSL_SUCCESS;
169 }
