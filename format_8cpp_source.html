18  openfsl::FAT32::getDefaultFormatOptions(const lba48_t fsSize) {
19  FormatOptions ret;
20  ret.volumeLabel = "NO NAME";
21 
22  auto tpoint = std::chrono::system_clock::now();
23  time_t ttenth = std::chrono::duration_cast<std::chrono::seconds>
24  (tpoint.time_since_epoch()).count() / 10;
25  ttenth %= 100;
26  time_t t = std::chrono::system_clock::to_time_t(tpoint);
27  std::tm currentTime = fsl_localtime(t);
28 
29  uint16_t tmp1 = (((uint8_t)currentTime.tm_mon + 1) << 8) +
30  (uint8_t)currentTime.tm_mday;
31  uint16_t tmp2 = ((uint8_t)currentTime.tm_sec << 8) + (uint8_t)ttenth;
32  ret.volumeSerial = (tmp1 + tmp2) << 16;
33 
34  tmp1 = ((uint8_t)currentTime.tm_hour << 8) + (uint8_t)currentTime.tm_min;
35  tmp2 = (uint16_t)currentTime.tm_year + 1900;
36  ret.volumeSerial += tmp1 + tmp2;
37 
38  std::cout << std::hex << ret.volumeSerial << std::dec << std::endl;
39 
40  ret.biosDriveNum = 0x80;
41 
42  if (fsSize < 532480)
43  ret.sectorsPerCluster = 1;
44  else if (fsSize < 16777216)
45  ret.sectorsPerCluster = 8;
46  else if (fsSize < 33554432)
47  ret.sectorsPerCluster = 16;
48  else if (fsSize < 67108864)
49  ret.sectorsPerCluster = 32;
50  else
51  ret.sectorsPerCluster = 64;
52 
53  ret.backupBSLocation = 6;
54  ret.fsinfoLocation = 1;
55  ret.reservedSectorCount = 32;
56 
57  ret.fatAreaCount = 2;
58  ret.fatMediaType = 0xF8;
59  ret.rootDirectoryCluster = 2;
60 
61  return ret;
62 }

64 error_t openfsl::FAT32::format(const lba48_t offset, const lba48_t size,
65  const FAT32::FormatOptions formatOptions, const bool fastFormat) {
66  BPBSector bpb;
67  FSINFOSector fsinfo;
68 
69  memset(&bpb, 0, sizeof(BPBSector));
70  memset(&fsinfo, 0, sizeof(FSINFOSector));
71 
72  // Set signatures
73  bpb.bpbVBRSignature = 0xAA55;
74  bpb.ebpbExtendedBootSign = 0x29;
75 
76  fsinfo.fsinfoSignature1 = 0x41615252;
77  fsinfo.fsinfoSignature2 = 0x61417272;
78  fsinfo.fsinfoSignature3 = 0xAA55;
79 
80  // Copy VBR boot codes
81  bpb.bpbJumpCode[0] = 0xEB;
82  bpb.bpbJumpCode[1] = 0x58;
83  bpb.bpbJumpCode[2] = 0x90;
84 
85  memcpy(bpb.bpbBootCode, _blob_vbrboot_bin_data, _blob_vbrboot_bin_len);
86 
87  // Copy volume OEM ID, Label, Volume ID, FS Type
88  if (formatOptions.volumeLabel.length() > 11) {
89  return OPENFSL_ERROR_TOO_LONG_VOLUME_LABEL;
90  }
91 
92  memcpy(bpb.bpbOEMName, "OPENFSL ", 8);
93 
94  memset(bpb.ebpbVolumeLabel, ' ', 11);
95  memcpy(bpb.ebpbVolumeLabel,
96  formatOptions.volumeLabel.c_str(),
97  formatOptions.volumeLabel.length());
98 
99  bpb.ebpbVolumeSerial = (uint32_t)formatOptions.volumeSerial;
100 
101  memcpy(bpb.ebpbFSType, "FAT32 ", 8);
102 
103  // Disk informations
104  if (formatOptions.biosDriveNum < 80) {
105  return OPENFSL_ERROR_FLOPPY_NOT_SUPPORTED;
106  }
107 
108  bpb.bpbMedia = 0xF8;
109  bpb.ebpbPhysicalDriveNum = formatOptions.biosDriveNum;
110  bpb.bpbBytesPerSector =
111  static_cast<uint16_t>(iod->getDiskParameter().bytesPerSector);
112  bpb.bpbSectorsPerTrack =
113  static_cast<uint16_t>(iod->getDiskParameter().sectorPerTrack);
114  bpb.bpbDiskHeads =
115  static_cast<uint16_t>(iod->getDiskParameter().headPerCylinder);
116 
117  // Cluster, sector informations
118  bpb.bpbHiddenSectors = (uint32_t)offset;
119  bpb.bpbTotalSector32 = (uint32_t)size;
120 
121  bpb.bpbSectorPerCluster = formatOptions.sectorsPerCluster;
122  bpb.bpbReservedSectors = (uint16_t)formatOptions.reservedSectorCount;
123 
124  if ((formatOptions.backupBSLocation >= formatOptions.reservedSectorCount) ||
125  (formatOptions.fsinfoLocation >= formatOptions.reservedSectorCount)) {
126  return OPENFSL_ERROR_INVALID_FORMAT_OPTIONS;
127  }
128 
129  bpb.ebpbBSBackupSector = (uint16_t)formatOptions.backupBSLocation;
130  bpb.ebpbFSInfoSector = (uint16_t)formatOptions.fsinfoLocation;
131 
132  // FAT Informations
133  lba48_t tmpval1 = size - formatOptions.reservedSectorCount;
134  lba48_t tmpval2 =
135  128 * formatOptions.sectorsPerCluster + formatOptions.fatAreaCount;
136 
137  bpb.bpbFATCount = formatOptions.fatAreaCount;
138  bpb.ebpbSectorsPerFAT32 = (uint32_t)((tmpval1 + tmpval2 - 1) / tmpval2);
139  bpb.ebpbRootDirectoryCluster = formatOptions.rootDirectoryCluster;
140 
141  cluster_t clusterCount = static_cast<cluster_t>
142  ((size - (formatOptions.reservedSectorCount +
143  bpb.ebpbSectorsPerFAT32 * formatOptions.fatAreaCount))
144  / formatOptions.sectorsPerCluster);
145 
146  if (bpb.ebpbRootDirectoryCluster >= clusterCount)
147  return OPENFSL_ERROR_INVALID_FORMAT_OPTIONS;
148 
149  fsinfo.fsinfoFreeCluster = clusterCount - 1;
150  if (bpb.ebpbRootDirectoryCluster == 1)
151  fsinfo.fsinfoNextFree = 2;
152  else
153  fsinfo.fsinfoNextFree = 0;
154 
155  __hexdump(reinterpret_cast<uint8_t*>(&bpb), 0, 512);
156  __hexdump(reinterpret_cast<uint8_t*>(&fsinfo), 0, 512);
157 
158  error_t result;
159 
160  // Zerofill disk
161  if (!fastFormat) {
162  char* emptySector = new char[iod->getDiskParameter().bytesPerSector];
163  memset(emptySector, 0, iod->getDiskParameter().bytesPerSector);
164 
165  for (lba48_t i = offset + formatOptions.reservedSectorCount +
166  bpb.ebpbSectorsPerFAT32 * formatOptions.fatAreaCount;
167  i < offset + size;
168  i++) {
169  result = iod->writeSector(emptySector, i, 1);
170  if (result)
171  return result;
172  }
173 
174  delete[] emptySector;
175  }
176 
177  // Disk operations
178  result = iod->writeSector(&bpb, offset, 1);
179  if (result)
180  return result;
181  result = iod->writeSector(&fsinfo,
182  offset + formatOptions.fsinfoLocation, 1);
183  if (result)
184  return result;
185 
186  // FAT Area initialization
187  Sector fatArea(bpb.ebpbSectorsPerFAT32,
188  iod->getDiskParameter().bytesPerSector);
189  memset(fatArea.getData(), 0, fatArea.getSize());
190  fatArea.getDataCast<uint32_t>()[0] =
191  0x0FFFFF00 + formatOptions.fatMediaType;
192  fatArea.getDataCast<uint32_t>()[1] = 0x0FFFFFFF;
193  fatArea.getDataCast<uint32_t>()[formatOptions.rootDirectoryCluster] =
194  0x0FFFFFFF;
195 
196  for (int i = 0; i < formatOptions.fatAreaCount; i++) {
197  result = iod->writeSector(fatArea.getDataCast<uint8_t>(),
198  offset + formatOptions.reservedSectorCount +
199  bpb.ebpbSectorsPerFAT32 * i,
200  bpb.ebpbSectorsPerFAT32);
201  if (result)
202  return result;
203  }
204 
205  uint8_t* emptycluster =
206  new uint8_t[iod->getDiskParameter().bytesPerSector *
207  formatOptions.sectorsPerCluster];
208  memset(emptycluster, 0, iod->getDiskParameter().bytesPerSector *
209  formatOptions.sectorsPerCluster);
210 
211  result = iod->writeSector(emptycluster,
212  offset + formatOptions.reservedSectorCount +
213  bpb.ebpbSectorsPerFAT32 * formatOptions.fatAreaCount +
214  formatOptions.sectorsPerCluster *
215  (formatOptions.rootDirectoryCluster - 2),
216  formatOptions.sectorsPerCluster);
217 
218  delete[] emptycluster;
219 
220  return OPENFSL_SUCCESS;
221 }
