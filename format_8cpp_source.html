14 FormatOptions FS_FAT32::getDefaultFormatOptions(lba48_t fsSize) {
15  FormatOptions ret;

18  std::chrono::system_clock::time_point tpoint = std::chrono::system_clock::now();
19  time_t ttenth = std::chrono::duration_cast<std::chrono::seconds>(tpoint.time_since_epoch()).count() / 10;
20  ttenth %= 100;

22  std::tm currentTime = fsl_localtime(t);
23 
24  uint16_t tmp1 = (((uint8_t)currentTime.tm_mon + 1) << 8) + (uint8_t)currentTime.tm_mday;
25  uint16_t tmp2 = ((uint8_t)currentTime.tm_sec << 8) + (uint8_t)ttenth;

27

30  ret.volumeSerial += tmp1 + tmp2;
31 
32  ret.biosDriveNum = 0x80;
33 
34  if (fsSize < 131072)
35  ret.sectorsPerCluster = 1;
36  else if (fsSize < 262144)

45  ret.sectorsPerCluster = 32;
46
