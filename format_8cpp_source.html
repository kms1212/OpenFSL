19 FS_FAT32::FormatOptions FS_FAT32::getDefaultFormatOptions(lba48_t fsSize) {
20  FormatOptions ret;
21  ret.volumeLabel = "NO NAME";
22 
23  std::chrono::system_clock::time_point tpoint = std::chrono::system_clock::now();
24  time_t ttenth = std::chrono::duration_cast<std::chrono::seconds>(tpoint.time_since_epoch()).count() / 10;
25  ttenth %= 100;
26  time_t t = std::chrono::system_clock::to_time_t(tpoint);
27  std::tm currentTime = fsl_localtime(t);
28 
29  uint16_t tmp1 = (((uint8_t)currentTime.tm_mon + 1) << 8) + (uint8_t)currentTime.tm_mday;
30  uint16_t tmp2 = ((uint8_t)currentTime.tm_sec << 8) + (uint8_t)ttenth;
31  ret.volumeSerial = (tmp1 + tmp2) << 16;
32 
33  tmp1 = ((uint8_t)currentTime.tm_hour << 8) + (uint8_t)currentTime.tm_min;
34  tmp2 = (uint16_t)currentTime.tm_year + 1900;
35  ret.volumeSerial += tmp1 + tmp2;
36 
37  ret.biosDriveNum = 0x80;
38 
39  if (fsSize < 532480)
40  ret.sectorsPerCluster = 1;
41  else if (fsSize < 16777216)
42  ret.sectorsPerCluster = 8;
43  else if (fsSize < 33554432)
44  ret.sectorsPerCluster = 16;
45  else if (fsSize < 67108864)
46  ret.sectorsPerCluster = 32;
47  else
48  ret.sectorsPerCluster = 64;
49 
50  ret.backupBSLocation = 6;
51  ret.fsinfoLocation = 1;
52  ret.reservedSectorCount = 32;
53 
54  ret.fatAreaCount = 2;
55  ret.fatMediaType = 0xF8;
56  ret.rootDirectoryCluster = 2;
57 
58  return ret;
59 }

61 void __hexdump(const uint8_t* p, size_t offset, size_t len)
62 {
63  std::ios init(NULL);
64  init.copyfmt(std::cout);
65 
66  size_t address = 0;
67  size_t row = 0;
68 
69  std::cout << std::hex << std::setfill('0');
70  while (1) {
71  if (address >= len) break;
72  size_t nread = ((len - address) > 16) ? 16 : (len - address);
73 
74  // Show the address
75  std::cout << std::right << std::setw(8) << address + offset;
76 
77  // Show the hex codes
78  for (size_t i = 0; i < 16; i++)
79  {
80  if (i % 8 == 0) std::cout << ' ';
81  if (i < nread)
82  std::cout << ' ' << std::setw(2) << (int)p[16 * row + i + offset];
83  else
84  std::cout << " ";
85  }
86 
87  // Show printable characters
88  std::cout << " ";
89  for (size_t i = 0; i < nread; i++)
90  {
91  uint8_t ch = p[16 * row + i + offset];
92  if (ch < 32 || ch > 126) std::cout << '.';
93  else std::cout << ch;
94  }
95 
96  std::cout << "\n";
97  address += 16;
98  row++;
99  }
100 
101  std::cout.flush();
102  std::cout.copyfmt(init);
103 }

105 error_t FS_FAT32::format(lba48_t offset, lba48_t size, FS_FAT32::FormatOptions formatOptions, bool fastFormat) {
106  BPBSector bpb = { 0 };
107  FSINFOSector fsinfo = { 0 };
108 
109  // Set signatures
110  bpb.bpbVBRSignature = 0xAA55;
111  bpb.ebpbExtendedBootSign = 0x29;
112 
113  fsinfo.fsinfoSignature1 = 0x41615252;
114  fsinfo.fsinfoSignature2 = 0x61417272;
115  fsinfo.fsinfoSignature3 = 0xAA55;
116 
117  // Copy VBR boot codes
118  bpb.bpbJumpCode[0] = 0xEB;
119  bpb.bpbJumpCode[1] = 0x58;
120  bpb.bpbJumpCode[2] = 0x90;
121 
122  memcpy(bpb.bpbBootCode, _blob_vbrboot_bin_data, _blob_vbrboot_bin_len);
123 
124  // Copy volume OEM ID, Label, Volume ID, FS Type
125  if (formatOptions.volumeLabel.length() > 11)
126  {
127  return OPENFSL_ERROR_VLABLONG; // Volume label is too long
128  }
129 
130  memcpy(bpb.bpbOEMName, "OPENFSL ", 8);
131 
132  memset(bpb.ebpbVolumeLabel, ' ', 11);
133  memcpy(bpb.ebpbVolumeLabel, formatOptions.volumeLabel.c_str(), formatOptions.volumeLabel.length());
134 
135  bpb.ebpbVolumeSerial = (uint16_t)formatOptions.volumeSerial;
136 
137  memcpy(bpb.ebpbFSType, "FAT32 ", 8);
138 
139  // Disk informations
140  if (formatOptions.biosDriveNum < 80)
141  {
142  return OPENFSL_ERROR_NOFLOPPY;
143  }
144 
145  bpb.bpbMedia = 0xF8;
146  bpb.ebpbPhysicalDriveNum = formatOptions.biosDriveNum;
147  bpb.bpbBytesPerSector = getDiskDevice()->getDiskParameter().bytesPerSector;
148  bpb.bpbSectorsPerTrack = getDiskDevice()->getDiskParameter().sectorPerTrack;
149  bpb.bpbDiskHeads = getDiskDevice()->getDiskParameter().headPerCylinder;
150 
151  // Cluster, sector informations
152  bpb.bpbHiddenSectors = (uint32_t)offset;
153  bpb.bpbTotalSector32 = (uint32_t)size;
154 
155  bpb.bpbSectorPerCluster = formatOptions.sectorsPerCluster;
156  bpb.bpbReservedSectors = (uint16_t)formatOptions.reservedSectorCount;
157 
158  if ((formatOptions.backupBSLocation >= formatOptions.reservedSectorCount) ||
159  (formatOptions.fsinfoLocation >= formatOptions.reservedSectorCount))
160  {
161  return OPENFSL_ERROR_INVFMTOPT;
162  }
163 
164  bpb.ebpbBSBackupSector = (uint16_t)formatOptions.backupBSLocation;
165  bpb.ebpbFSInfoSector = (uint16_t)formatOptions.fsinfoLocation;
166 
167  // FAT Informations
168  lba48_t tmpval1 = size - formatOptions.reservedSectorCount;
169  lba48_t tmpval2 = 128 * formatOptions.sectorsPerCluster + formatOptions.fatAreaCount;
170 
171  bpb.bpbFATCount = formatOptions.fatAreaCount;
172  bpb.ebpbSectorsPerFAT32 = (uint32_t)((tmpval1 + tmpval2 - 1) / tmpval2);
173  bpb.ebpbRootDirectoryCluster = formatOptions.rootDirectoryCluster;
174 
175  cluster_t clusterCount = (cluster_t)
176  ((size - (formatOptions.reservedSectorCount + bpb.ebpbSectorsPerFAT32 * formatOptions.fatAreaCount))
177  / formatOptions.sectorsPerCluster);
178 
179  fsinfo.fsinfoFreeCluster = clusterCount;
180 
181  __hexdump((uint8_t*)&bpb, 0, 512);
182  __hexdump((uint8_t*)&fsinfo, 0, 512);
183 
184  // Disk operations
185  error_t result = dd->writeDisk(&bpb, offset, 1);
186  if (result)
187  return result;
188  result = dd->writeDisk(&fsinfo, offset + formatOptions.fsinfoLocation, 1);
189  if (result)
190  return result;
191 
192  if (!fastFormat) { //Zerofill disk
193  char* emptySector = new char[dd->getDiskParameter().bytesPerSector];
194 
195  for (lba48_t i = offset + formatOptions.reservedSectorCount + bpb.ebpbSectorsPerFAT32 * formatOptions.fatAreaCount;
196  i < offset + size;
197  i++) {
198  result = dd->writeDisk(emptySector, i, 1);
199  if (result)
200  return result;
201  }
202 
203  delete[] emptySector;
204  }
205 
206  return OPENFSL_SUCCESS;
207 }
