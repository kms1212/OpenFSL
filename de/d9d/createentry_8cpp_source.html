16 size_t __getdigitcount(size_t n) {
17  size_t ret;
18  for (ret = 1;; ret++) {
19  if (n >= 10) {
20  ret++;
21  n /= 10;
22  } else {
23  break;
24  }
25  }
26  return ret;
27 }

35  return OPENFSL_ERROR_TOO_LARGE_FILE_SIZE;
36

77  for (auto & c : filename) c = static_cast<char>(toupper(c));
78

100

102  if (i == totalEntryCount) {
103  result = lcluster.addCluster(1);
104  if (result) {
105  currentPath = tempPath;
106  currentCluster = tempCluster;
107  return result;
108  }
109  totalEntryCount = lcluster.getClusterList()->size() * 16;
110  }
111

113  result = lcluster.read(&deletionMark, i * 32, 1);
114  if (result) {
115  currentPath = tempPath;
116  currentCluster = tempCluster;
117  return result;
118  }
119 
120  if ((deletionMark == 0xE5) || (deletionMark == 0x00))
121  blankPlaceCount++;
122  else
123  blankPlaceCount = 0;
124 
125  if (blankPlaceCount >= requiredEntrySize) {
126  entryPlace = i;
127  break;
128  }
129  }

133  fileEntry.fileAttr = createFileInfo.fileAttr;

135  fileEntry.fileLocationLow = (uint16_t)createFileInfo.fileLocation;

153

158  filename_sfn
159  .push_back((c < 0x80) && (c != '.') ?

168  sfnDuplicateCountString.c_str(),

173  // every non-ASCII character to underscore
174  if (dotpos != filename16.size()) {
175  std::string fileext_sfn;
176  size_t extsize = filename16.size() - dotpos;
177  for (auto c :
178  filename16.substr(dotpos + 1, extsize > 3 ? 3 : extsize))

220  }

251  currentCluster = tempCluster;
252  return result;
253  }
254

257 
258  requiredEntrySize--;
259  }
260  } else { // If file name is SFN
261  std::string createFilename = createFileInfo.fileName;
262  for (auto & c : createFilename) c = static_cast<char>(toupper(c));
263  std::string filename_str = createFilename.substr(0, dotpos);
264  std::string fileext_str = "";

268  memset(fileEntry.fileExt, ' ', 3);
