38 uint32_t FS_FAT32::LinkedCluster::getNextCluster(uint32_t cluster)
39 {
40  uint32_t temp = cluster & ~(0xF << 28); // Ignore first 4 bits
41  temp = ((uint32_t*)(fileSystem->getFatArea()->getData()))[temp] & ~(0xF << 28); // Get next cluster

43  return 0xFFFFFFF8;
