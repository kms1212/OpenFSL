18 int openfsl::MBR::initialize() {

20 
21  Sector sector_lba0(1, bd->getDiskParameter().bytesPerSector);
22  // Read partition table sector
23  int errorcode = bd->readSector(sector_lba0.getData(), 0, 1);
24  if (errorcode)
25  return errorcode;

27  memcpy(&table, sector_lba0.getData(), sizeof(PartitionTable));
28 
29  if (leToSystem<uint16_t>(table.mbrSignature) != 0xAA55) // Check MBR signature
30  return OPENFSL_ERROR_INVALID_SIGNATURE;
31 
32  for (int i = 0; i < 256; i++) {
33  PartitionEntry currentEntry;
34 
35  if (i < 4) {
36  currentEntry = table.mbrPartitionEntry[i];
37  currentEntry.entryStartingLBAAddr =
38  leToSystem<uint32_t>(currentEntry.entryStartingLBAAddr);
39  currentEntry.entryLBASize =
40  leToSystem<uint32_t>(currentEntry.entryLBASize);
41 
42  if (currentEntry.entryPartitionType == 0)
43  break;
44 
45  // Exended Partition
46  if ((currentEntry.entryPartitionType ==
47  PartitionType::ExtendedPartitionCHS) ||
48  (currentEntry.entryPartitionType ==
49  PartitionType::ExtendedPartitionLBA)) {
50  // If partition table created without LBA Addressing
51  if (currentEntry.entryStartingLBAAddr == 0) {
52  } else { // If partition table created with LBA Addressing
53  base_addr = currentEntry.entryStartingLBAAddr;
54 
55  // Read extended partition table
56  errorcode =
57  bd->readSector(sector_lba0.getData(), base_addr, 1);
58  if (errorcode)
59  return errorcode;
60

63  mbrPartitionEntry[0];
64

66  break;

69  partitionList.push_back(currentEntry);

73  break;
74  } else {
75  // If partition table created without LBA Addressing
76  if (sector_lba0.getDataCast<PartitionTable>()->

78  // add lba-converted chs address to base_addr

88  }
89  } else {

92  if (errorcode)

142

144  CHS chs;

148  bd->getDiskParameter().sectorPerTrack,

150

154  }
