46  {
47 
48  }

52  if (errorcode)

122 void TB_MBR::registerFilesystem(FileSystem* fs, uint8_t index) {
123  if (partitionList.at(index).entryStartingLBAAddr == 0) // If partition table created without LBA Addressing
124  {
125  CHS chs;
126  uint32_t chsRaw = 0;

128  chs.cyl = (chsRaw & 0b111111111100000000000000) >> 16;
129  chs.head = chsRaw & 0b000000000000000011111111;
130  chs.sect = (chsRaw & 0b000000000011111100000000) >> 8;
131  fs->setFsLBAOffset(convertCHSToLBA(chs, 63, 16));
132  }
133  else { // If partition table created with LBA Addressing
134  fs->setFsLBAOffset(partitionList.at(index).entryStartingLBAAddr);
