21 int openfsl::MBR::initialize() {
22  lba48_t base_addr = 0;
23 
24  Sector sector_lba0(1, bd->getDiskParameter().bytesPerSector);
25  // Read partition table sector
26  int errorcode = bd->readSector(sector_lba0.getData(), 0, 1);
27  if (errorcode)
28  return errorcode;
29  table = new PartitionTable();
30  memcpy(table, sector_lba0.getData(), sizeof(PartitionTable));
31 
32  if (table->mbrSignature != 0xAA55) // Check MBR signature
33  return OPENFSL_ERROR_INVSIGN;
34 
35  for (int i = 0; i < 256; i++) {
36  PartitionEntry currentEntry;
37 
38  if (i < 4) {
39  currentEntry = table->mbrPartitionEntry[i];
40 
41  if (currentEntry.entryPartitionType == 0)
42  break;
43 
44  // Exended Partition
45  if ((currentEntry.entryPartitionType == PartitionType::ExtPart_S) ||
46  (currentEntry.entryPartitionType == PartitionType::ExtPart_L)) {
47  // If partition table created without LBA Addressing
48  if (currentEntry.entryStartingLBAAddr == 0) {
49  } else { // If partition table created with LBA Addressing
50  base_addr = currentEntry.entryStartingLBAAddr;
51 
52  // Read extended partition table
53  errorcode =
54  bd->readSector(sector_lba0.getData(), base_addr, 1);
55  if (errorcode)
56  return errorcode;
57 
58  currentEntry =
59  sector_lba0.getDataCast<PartitionTable*>()->
60  mbrPartitionEntry[0];
61 
62  if (currentEntry.entryPartitionType == 0)
63  break;
64 
65  currentEntry.entryStartingLBAAddr += (uint32_t)base_addr;
66  partitionList.push_back(currentEntry);
67 
68  if (sector_lba0.getDataCast<PartitionTable*>()->
69  mbrPartitionEntry[1].entryPartitionType == 0) {
70  break;
71  } else {
72  // If partition table created without LBA Addressing
73  if (sector_lba0.getDataCast<PartitionTable*>()->
74  mbrPartitionEntry[1].entryStartingLBAAddr == 0) {
75  // add lba-converted chs address to base_addr
76  } else { // With LBA Addressing
77  base_addr +=
78  sector_lba0.getDataCast<PartitionTable*>()->
79  mbrPartitionEntry[1].entryStartingLBAAddr;
80  }
81  }
82  }
83  } else { // Primary partition
84  partitionList.push_back(currentEntry);
85  }
86  } else {
87  // Read extended partition table
88  errorcode = bd->readSector(sector_lba0.getData(), base_addr, 1);
89  if (errorcode)
90  return errorcode;
91 
92  PartitionEntry currentEntry =
93  sector_lba0.getDataCast<PartitionTable*>()->
94  mbrPartitionEntry[0];
95 
96  currentEntry.entryStartingLBAAddr += (uint32_t)base_addr;
97  partitionList.push_back(currentEntry);
98 
99  if (sector_lba0.getDataCast<PartitionTable*>()->

101  break;
102  } else {
103  // If partition table created without LBA Addressing

105  mbrPartitionEntry[1].entryStartingLBAAddr == 0) {
106  // add lba-converted chs address to base_addr

118 std::vector<openfsl::FileSystemType> openfsl::MBR::getPartitionTypes() {
119  std::vector<FileSystemType> ret;
120  for (size_t i = 0; i < partitionList.size(); i++) {
121  FileSystemType fs;
122  if (partitionList[i].entryPartitionType == PartitionType::FAT12)
123  fs = FileSystemType::FAT12;
124  else if (partitionList[i].entryPartitionType ==
125  PartitionType::FAT16_LBA)
126  fs = FileSystemType::FAT16;
127  else if ((partitionList[i].entryPartitionType ==
128  PartitionType::FAT32) ||
129  (partitionList[i].entryPartitionType ==
130  PartitionType::FAT32_LBA))
131  fs = FileSystemType::FAT32;
132  else if (partitionList[i].entryPartitionType == PartitionType::NTFS)
133  fs = FileSystemType::NTFS;
134 
135  ret.push_back(fs);
136  }
