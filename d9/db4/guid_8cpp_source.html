63  char stdGuidString[37];
64 
65  if (guidString[0] == '{') {
66  if ((strnlen(guidString, 40) != 38) &&
67  (guidString[37] != '}')) {
68  throw std::invalid_argument("Invalid GUID string format");
69  }
70 
71  strncpy(stdGuidString, guidString + 1, 36);
72  } else {
73  if (strnlen(guidString, 40) != 36) {
74  throw std::invalid_argument("Invalid GUID string format");
75  }
76 
77  strncpy(stdGuidString, guidString, 36);
78  }
79 
80  if ((stdGuidString[8] != '-') ||
81  (stdGuidString[13] != '-') ||
82  (stdGuidString[18] != '-') ||
83  (stdGuidString[23] != '-')) {
84  throw std::invalid_argument("Invalid GUID string format");
85  }
86 
87  stdGuidString[8] = 0;
88  stdGuidString[13] = 0;
89  stdGuidString[18] = 0;
90  stdGuidString[23] = 0;
91  stdGuidString[36] = 0;
92 
93  for (char c : stdGuidString) {
94  if (!(c >= '0' && c <= '9') &&
95  !(c >= 'A' && c <= 'F') &&
96  !(c >= 'a' && c <= 'f') &&
97  c != 0) {
98  throw std::invalid_argument("Invalid GUID string format");
99  }
100  }
101 
102  guid1 = (uint32_t)strtoul(stdGuidString, NULL, 16);
103  guid2 = (uint16_t)strtoul(stdGuidString + 9, NULL, 16);
104  guid3 = (uint16_t)strtoul(stdGuidString + 14, NULL, 16);

153 openfsl::GUID openfsl::GUID::generateGuid4() {
154  std::random_device rd;
155  std::mt19937 rng(rd());
156  std::mt19937_64 rng64(rd());
157 
158  std::uniform_int_distribution<uint16_t> g3dist(0, 0xFFF);
159  uint16_t guid3 = g3dist(rng);
160  guid3 += 0x4000;
161 
162  std::uniform_int_distribution<uint16_t> g4dist(0, 0x3FFF);
163  uint16_t guid4 = g4dist(rng);
164  guid4 += 0x8000;
165 
166  return openfsl::GUID(rng(), rng(), guid3, guid4, rng64());
167 }
