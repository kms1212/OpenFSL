46 int openfsl::convertLBAToCHS(const lba48_t lba, CHS* chs,
47  const uint8_t sectPerTra, const uint8_t headPerCyl) {
48  if (lba > 16515135)
49  return OPENFSL_ERROR_CANNOT_CONVERT_TO_CHS;
50  chs->cyl = lba / (headPerCyl * sectPerTra);
51  chs->head = (lba / sectPerTra) % headPerCyl;
52  chs->sect = (lba % sectPerTra) + 1;
53  return 0;
