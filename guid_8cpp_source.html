168 openfsl::GUID openfsl::GUID::generateGuid4() {
169  std::random_device rd;
170  std::mt19937 rng(rd());
171  std::mt19937_64 rng64(rd());
172 
173  std::uniform_int_distribution<uint16_t> g3dist(0, 0xFFF);
174  uint16_t guid3 = g3dist(rng);
175  guid3 += 0x4000;
176 
177  std::uniform_int_distribution<uint16_t> g4dist(0, 0x3FFF);
178  uint16_t guid4 = g4dist(rng);
179  guid4 += 0x8000;
180 
181  return openfsl::GUID(rng(), rng(), guid3, guid4, rng64());
182 }
