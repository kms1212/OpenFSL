135  openfsl::fat32::ClusterChain::getNextCluster(const cluster_t cluster) {
136  cluster_t temp = cluster & ~(0xF << 28); // Ignore first 4 bits
137  // Get next cluster
138  temp = (reinterpret_cast<cluster_t*>
139  (fileSystem->getFatArea()->getData()))[temp] & ~(0xF << 28);
140  if (0xFFFFFFF - temp <= 7) // Check bad cluster or cluster end
141  return 0xFFFFFFF8;
142  else if (0xFFFFFFF - temp == 8)
143  return 0xFFFFFFF7;
144  else
