14 int FS_FAT32::LinkedCluster::addCluster(cluster_t addCount)
15 {
16  cluster_t nextFreeCluster = fileSystem->getNextFreeCluster();
17  std::set<lba48_t> modifiedSectorList;
18  cluster_t* fatArea = fileSystem->getFatArea()->getDataCast<cluster_t*>();
19 
20  size_t bytespersector = fileSystem->getDiskDevice()->getDiskParameter().bytesPerSector;
21 
22  for (size_t i = 0; i < addCount; i++)
23  {
24  if (nextFreeCluster == 0) {
25  return OPENFSL_ERROR_ALLOC;
26  }
27 
28  cluster_t prevLastCluster = clusterList.at(clusterList.size() - 1);
29 
30  fatArea[prevLastCluster] = nextFreeCluster;
31  fatArea[nextFreeCluster] = 0x0FFFFFFF;
32 
33  updateClusterList();
34 
35  uint8_t* blankSector = new uint8_t[bytespersector];
36  memset(blankSector, 0, bytespersector);
37  for (lba48_t i = 0; i < bytespersector; i++)
38  fileSystem->getDiskDevice()->writeDisk(blankSector,
39  fileSystem->getActualLBA(clusterToLBA(nextFreeCluster) + i), 1);
40  delete[] blankSector;
41 
42  modifiedSectorList.insert(fileSystem->fatAreaIndexToSector(prevLastCluster));
43  modifiedSectorList.insert(fileSystem->fatAreaIndexToSector(nextFreeCluster));
44 
45  nextFreeCluster = fileSystem->getNextFreeCluster();
46  }
47 
48  for (std::set<lba48_t>::iterator i = modifiedSectorList.begin();
49  i != modifiedSectorList.end(); ++i)
50  {
51  fileSystem->getDiskDevice()->writeDisk((uint8_t*)fatArea + (*i - fileSystem->getBPB()->bpbReservedSectors) * bytespersector,
52  fileSystem->getActualLBA(*i), 1);
53  }
54 
55  // Write result to FSINFO structure
56  fileSystem->fsinfo->fsinfoFreeCluster -= addCount;
57  fileSystem->fsinfo->fsinfoNextFree = nextFreeCluster;
58  int errcode = fileSystem->updateFsInfo();
59  if (errcode)
60  return errcode;
61 
62  return 0;
63 }
