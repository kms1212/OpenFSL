20  size_t bytespersector = fileSystem->getDiskDevice()->getDiskParameter().bytesPerSector;
21 
22  for (size_t i = 0; i < addCount; i++)
23  {
24  if (nextFreeCluster == 0) {

37  memset(blankSector, 0, bytespersector);
38  for (size_t i = 0; i < bytespersector; i++)
39  fileSystem->getDiskDevice()->writeDisk(blankSector,
40  fileSystem->getActualLBA(clusterToLBA(nextFreeCluster) + i), 1);
41  delete[] blankSector;
42

44  modifiedSectorList.insert((uint32_t)fileSystem->fatAreaIndexToSector(nextFreeCluster));

55
