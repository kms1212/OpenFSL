114 error_t openfsl::GPT::initialize() {
115  error_t result = protMbr->initialize();
116  if (result)
117  return result;
118 
119  std::vector<MBR::PartitionInfo> fsList;
120  result = protMbr->getPartitionInfo(&fsList);
121  if (result)
122  return result;
123  tableOffset = fsList.at(0).partOffset;
124 
125  // Check if GPT entry is a partition of MBR
126  if (fsList.size() != 1 ||
127  fsList[0].partType != openfsl::MBR::PartitionType::GPTProtective)
128  return OPENFSL_ERROR_INVALID_SIGNATURE;
129 
130  // Read first sector
131  Sector headerSector(1, bd->getDiskParameter().bytesPerSector);
132  result = bd->readSector(headerSector.getData(), tableOffset, 1);
133  if (result)
134  return result;
135 
136  // Check signatures and CRC
137  memcpy(&gptHeader, headerSector.getData(), sizeof(TableHeader));
138 
139  if (memcmp("EFI PART", gptHeader.gptSignature, 8) != 0)
140  return OPENFSL_ERROR_INVALID_SIGNATURE;
141 
142  uint32_t origCrc32 = gptHeader.gptHeaderCrc32;
143  gptHeader.gptHeaderCrc32 = 0;
144 
145  uint32_t calculatedCrc32 =
146  crc32(0, reinterpret_cast<uint8_t*>(&gptHeader),
147  gptHeader.gptHeaderSize);
148 
149  if (origCrc32 != calculatedCrc32)
150  return OPENFSL_ERROR_INVALID_SIGNATURE;
151 
152  diskGuid.setGUID(gptHeader.gptDiskGuid);
153 
154  return 0;
155 }

157 error_t openfsl::GPT::getPartitionInfo(
158  std::vector<openfsl::GPT::PartitionInfo>* buf) {
159  lba48_t partitionEntryTableSize =
160  (gptHeader.gptPartitionEntryCount *
161  gptHeader.gptPartitionEntrySize /
162  bd->getDiskParameter().bytesPerSector) + 1;
163 
164  Sector partitionListSector(
165  partitionEntryTableSize, bd->getDiskParameter().bytesPerSector);
166  error_t result = bd->readSector(partitionListSector.getData(),
167  gptHeader.gptPartitionEntryStart, partitionEntryTableSize);
168  if (result)
169  return result;
170 
171  for (uint32_t i = 0; i < gptHeader.gptPartitionEntryCount; i++) {
172  PartitionEntry* currentPartitionEntry =
173  reinterpret_cast<PartitionEntry*>(
174  partitionListSector.getDataCast<uint8_t>() +
175  (i * gptHeader.gptPartitionEntrySize));
176 
177  GUID partitionTypeGuid(currentPartitionEntry->entryPartitionTypeGuid);
178  GUID partitionGuid(currentPartitionEntry->entryPartitionGuid);
179 
180  if (partitionTypeGuid == GUID()) {
181  break;
182  }
183  std::unordered_map<openfsl::GUID, PartitionType, openfsl::GUIDHash>::iterator it =
184  partitionGuidMap.find(partitionTypeGuid);
185 
186  PartitionInfo pinfo;
187 
188  if (it != partitionGuidMap.end())
189  pinfo.partType = static_cast<PartitionType>((*it).second);
190  pinfo.partAttribute = currentPartitionEntry->entryAttributeFlags;
191  pinfo.partOffset = currentPartitionEntry->entryPartitionStart;
192  pinfo.partSize =
193  currentPartitionEntry->entryPartitionEnd - pinfo.partOffset;
194 
195  buf->push_back(pinfo);
196  }
197 
198  return OPENFSL_SUCCESS;
199 }
